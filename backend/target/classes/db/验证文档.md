```sql
-- 预期结果：1条记录（admin用户）

-- 验证管理员角色关联
SELECT u.username, r.role_name FROM users u 
JOIN user_roles ur ON u.id = ur.user_id 
JOIN roles r ON ur.role_id = r.id 
WHERE u.username = 'admin';
-- 预期结果：admin用户拥有超级管理员角色

-- 验证角色模型映射数据
SELECT role_name, selection_strategy, is_enabled FROM role_model_mappings;
-- 预期结果：至少1条映射记录（普通用户角色）
```
